class BiLSTMTagger(nn.Module):
    """2-layer Bidirectional LSTM for sequence labeling"""
    
    def __init__(self, vocab_size, embedding_dim, hidden_dim, num_tags, 
                 pretrained_embeddings=None, freeze_embeddings=False, 
                 dropout=0.5, use_crf=False, pad_idx=0):
        super(BiLSTMTagger, self).__init__()
        
        self.use_crf = use_crf
        self.hidden_dim = hidden_dim
        self.pad_idx = pad_idx
        
        # Embedding layer
        self.embedding = nn.Embedding(vocab_size, embedding_dim, padding_idx=pad_idx)
        if pretrained_embeddings is not None:
            self.embedding.weight.data.copy_(torch.from_numpy(pretrained_embeddings))
        if freeze_embeddings:
            self.embedding.weight.requires_grad = False
        
        # 2-layer Bidirectional LSTM
        self.lstm = nn.LSTM(
            embedding_dim, 
            hidden_dim // 2,  # Divide by 2 because bidirectional
            num_layers=2,
            bidirectional=True,
            batch_first=True,
            dropout=dropout
        )
        
        # Dropout
        self.dropout = nn.Dropout(dropout)
        
        # Output layer
        self.fc = nn.Linear(hidden_dim, num_tags)
        
        # CRF layer (optional, for NER)
        if use_crf:
            self.crf = CRF(num_tags, pad_tag_idx=num_tags-1)
    
    def forward(self, word_ids, seq_lens, tags=None):
        """
        Args:
            word_ids: (batch, max_len)
            seq_lens: (batch,) actual sequence lengths
            tags: (batch, max_len) - only for training with CRF
        Returns:
            logits or loss
        """
        batch_size, max_len = word_ids.shape
        
        # Embeddings
        embedded = self.embedding(word_ids)  # (batch, max_len, emb_dim)
        
        # Pack padded sequences for LSTM
        packed = nn.utils.rnn.pack_padded_sequence(
            embedded, seq_lens.cpu(), batch_first=True, enforce_sorted=False
        )
        
        # BiLSTM
        lstm_out, _ = self.lstm(packed)
        
        # Unpack
        lstm_out, _ = nn.utils.rnn.pad_packed_sequence(lstm_out, batch_first=True)
        
        # Dropout
        lstm_out = self.dropout(lstm_out)
        
        # Linear layer
        emissions = self.fc(lstm_out)  # (batch, max_len, num_tags)
        
        if self.use_crf:
            # Create mask
            mask = torch.arange(max_len).expand(batch_size, max_len).to(word_ids.device)
            mask = mask < seq_lens.unsqueeze(1)
            mask = mask.float()
            
            if tags is not None:
                # Training: return CRF loss
                return self.crf(emissions, tags, mask)
            else:
                # Inference: return Viterbi decoding
                return self.crf.viterbi_decode(emissions, mask)
        else:
            return emissions

def train_model(model, train_loader, val_loader, optimizer, num_epochs, 
                device, use_crf=False, patience=5):
    """Train sequence labeling model"""
    
    train_losses = []
    val_losses = []
    val_f1_scores = []
    best_val_f1 = 0
    patience_counter = 0
    
    if not use_crf:
        criterion = nn.CrossEntropyLoss(ignore_index=model.pad_idx)
    
    for epoch in range(num_epochs):
        # Training
        model.train()
        epoch_train_loss = 0
        
        for batch in tqdm(train_loader, desc=f"Epoch {epoch+1}/{num_epochs}"):
            word_ids = batch['word_ids'].to(device)
            tag_ids = batch['tag_ids'].to(device)
            seq_lens = batch['seq_len']
            
            optimizer.zero_grad()
            
            if use_crf:
                loss = model(word_ids, seq_lens, tag_ids)
            else:
                outputs = model(word_ids, seq_lens)
                # Reshape for cross-entropy
                loss = criterion(outputs.view(-1, outputs.size(-1)), tag_ids.view(-1))
            
            loss.backward()
            optimizer.step()
            
            epoch_train_loss += loss.item()
        
        avg_train_loss = epoch_train_loss / len(train_loader)
        train_losses.append(avg_train_loss)
        
        # Validation
        model.eval()
        epoch_val_loss = 0
        all_preds = []
        all_labels = []
        
        with torch.no_grad():
            for batch in val_loader:
                word_ids = batch['word_ids'].to(device)
                tag_ids = batch['tag_ids'].to(device)
                seq_lens = batch['seq_len']
                
                if use_crf:
                    loss = model(word_ids, seq_lens, tag_ids)
                    predictions = model(word_ids, seq_lens)
                    # Flatten predictions
                    for i, pred_seq in enumerate(predictions):
                        all_preds.extend(pred_seq)
                        all_labels.extend(tag_ids[i, :len(pred_seq)].cpu().numpy())
                else:
                    outputs = model(word_ids, seq_lens)
                    loss = criterion(outputs.view(-1, outputs.size(-1)), tag_ids.view(-1))
                    predictions = torch.argmax(outputs, dim=-1)
                    
                    # Get valid predictions (exclude padding)
                    for i in range(len(seq_lens)):
                        seq_len = seq_lens[i]
                        all_preds.extend(predictions[i, :seq_len].cpu().numpy())
                        all_labels.extend(tag_ids[i, :seq_len].cpu().numpy())
                
                epoch_val_loss += loss.item()
        
        avg_val_loss = epoch_val_loss / len(val_loader)
        val_losses.append(avg_val_loss)
        
        # Compute F1
        from sklearn.metrics import f1_score
        val_f1 = f1_score(all_labels, all_preds, average='macro')
        val_f1_scores.append(val_f1)
        
        print(f"Epoch {epoch+1}: Train Loss={avg_train_loss:.4f}, Val Loss={avg_val_loss:.4f}, Val F1={val_f1:.4f}")
        
        # Early stopping
        if val_f1 > best_val_f1:
            best_val_f1 = val_f1
            patience_counter = 0
        else:
            patience_counter += 1
            if patience_counter >= patience:
                print(f"Early stopping triggered at epoch {epoch+1}")
                break
    
    return train_losses, val_losses, val_f1_scores

def assign_topic(doc, categories):
    """Assign topic based on keyword matching"""
    scores = {cat: 0 for cat in categories}
    for cat, keywords in categories.items():
        for keyword in keywords:
            if keyword in doc:
                scores[cat] += 1
    # Return category with highest score, or 'Politics' as default
    return max(scores.items(), key=lambda x: x[1])[0] if max(scores.values()) > 0 else 'Politics'
